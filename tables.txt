CREATE TABLE irradiance
(
    id_irr INT NOT NULL AUTO_INCREMENT,
    tanggal DATE NOT NULL,
    waktu VARCHAR(15) NOT NULL,
    value FLOAT NOT NULL,
    PRIMARY KEY (id_irr)
)

CREATE TABLE rompltd
(
    id_pltd INT NOT NULL AUTO_INCREMENT,
    unit INT NOT NULL,
    tanggal DATE NOT NULL,
    status INT NOT NULL,
    PRIMARY KEY (id_pltd)
)
